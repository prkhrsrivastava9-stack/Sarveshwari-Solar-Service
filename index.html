        async function handleFormSubmit(e) {

            e.preventDefault();

            const submitBtn = document.getElementById('submit-btn');

            const statusMsg = document.getElementById('form-status');

            

            if (recordCount >= 999) {

                statusMsg.textContent = "Inquiry limit reached. Please contact us directly via email.";

                return;

            }


            const name = document.getElementById('name').value;

            const email = document.getElementById('email').value;

            const message = document.getElementById('message').value;


            submitBtn.disabled = true;

            submitBtn.innerHTML = '<span class="animate-spin h-5 w-5 border-2 border-white border-t-transparent rounded-full mr-2"></span> Sending...';


            const result = await window.dataSdk.create({

                name,

                email,

                message,

            });

            submitBtn.innerHTML = '<span>Send Message</span><i data-lucide="send" class="w-4 h-4"></i>';

            initIcons();


            if (result.isOk) {

                statusMsg.textContent = "Thank you! We'll be in touch soon.";

                statusMsg.className = "text-center text-sm block text-green-600 mt-4";

                e.target.reset();

            } else {

                statusMsg.textContent = "Something went wrong. Please try again.";

                statusMsg.className = "text-center text-sm block text-red-600 mt-4";

            }

            

            setTimeout(() => statusMsg.classList.add('hidden'), 5000);

        }

        async function updateUI(newConfig) {

            const baseSize = config.font_size || defaultConfig.font_size;

            // Contrast text for primary buttons (simple logic)

            const isDarkAction = (hex) => {

                const r = parseInt(hex.slice(1,3), 16);

                const g = parseInt(hex.slice(3,5), 16);

                const b = parseInt(hex.slice(5,7), 16);

                return (r*0.299 + g*0.587 + b*0.114) < 186;

            };

            const btnTextColor = isDarkAction(config.primary_action) ? '#ffffff' : '#000000';

            primaryBtn.style.color = btnTextColor;

            submitBtn.style.color = btnTextColor;


            const secondaryBtn = document.getElementById('btn-secondary');

            secondaryBtn.style.borderColor = config.secondary_action;

            secondaryBtn.style.color = config.secondary_action;

            document.getElementById('hero-title').textContent = config.hero_title;

            document.getElementById('hero-subtitle').textContent = config.hero_subtitle;

            document.getElementById('service-1-name').textContent = config.service_1_title;

            document.getElementById('service-2-name').textContent = config.service_2_title;

            document.getElementById('service-3-name').textContent = config.service_3_title;

        }
